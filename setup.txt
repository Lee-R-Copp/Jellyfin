# Install Required Dependencies
  sudo apt install apt-transport-https software-properties-common gnupg curl -y

# Import the Jellyfin GPG Key
  curl -fsSL https://repo.jellyfin.org/jellyfin_team.gpg.key | sudo gpg --dearmor -o /usr/share/keyrings/jellyfin-archive-keyring.gpg

# Add the Jellyfin Repository
  echo "deb [signed-by=/usr/share/keyrings/jellyfin-archive-keyring.gpg] https://repo.jellyfin.org/ubuntu $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/jellyfin.list

# Update the Package Index
  apt update  

# Install Jellyfin
  apt install jellyfin -y

# Enable and Start the Jellyfin Service
  systemctl enable jellyfin
  systemctl start jellyfin

# Allow Jellyfin Through the Firewall
  ufw allow 8096/tcp
  ufw status

# Access the Web Interface
  http://<server_ip>:8096

# Add shared resource from TrueNAS (adjust uid and gid as needed)
  apt-get install cifs-utils
  mkdir -p /mnt/truenas
  mount -t cifs //192.168.199.5/plex /mnt/truenas -o username=YOUR_USERNAME,password=YOUR_PASSWORD,uid=$(id -u),gid=$(id -g)

# Make shared resource persistent (adjust uid and gid as needed)
  edit /root/samba.creds.truenas
    username=YOUR_USERNAME
    password=YOUR_PASSWORD
  chmod 600 /root/samba.creds.truenas
  edit /etc/fstab
    //192.168.199.5/share /mnt/truenas cifs credentials=/root/samba.creds.truenas,uid=0,gid=0,noperm 0 0
  reboot
